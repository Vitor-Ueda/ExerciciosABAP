           p_valor2 TYPE p DECIMALS 2.

    WRITE:'O resultado do valor', p_valor2, '/', p_valor1, '=', result.
  ENDIF.
  IF p_valor1 > p_valor2.
    result = p_valor1 / p_valor2.
    WRITE:'O resultado do valor', p_valor1, '/', p_valor2, '=', result.
  ENDIF.
  IF p_valor1 EQ p_valor2.
    result = p_valor1 * p_valor2.

  ENDIF.
ENDFORM.
